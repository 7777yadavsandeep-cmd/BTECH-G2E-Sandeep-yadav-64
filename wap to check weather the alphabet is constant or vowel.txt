

#include <stdio.h>

int main() {
    char ch;

    // Prompt the user for input
    printf("Enter an alphabet: ");
    
    // Read the character input from the user
    // A space before %c is used to consume any leading whitespace, including the newline from the previous enter key press.
    scanf(" %c", &ch);

    // Check if the character is a vowel
    if (ch == 'a' || ch == 'e' || ch == 'i' || ch == 'o' || ch == 'u' ||
        ch == 'A' || ch == 'E' || ch == 'I' || ch == 'O' || ch == 'U') {
        printf("%c is a vowel.\n", ch);
    }
    // Check if the character is an alphabet, but not a vowel (i.e., a consonant)
    else if ((ch >= 'a' && ch <= 'z') || (ch >= 'A' && ch <= 'Z')) {
        printf("%c is a consonant.\n", ch);
    }
    // If the character is neither a vowel nor a consonant, it is not an alphabet
    else {
        printf("%c is not an alphabet.\n", ch);
    }

    return 0;
}
